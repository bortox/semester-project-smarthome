194 uint8_t _lastMultiplier;
195 static constexpr uint8_t MS_PER_STEP = 4;
196
197 // Integer-based brightness multiplier (0-100)
198 static uint8_t& brightness_multiplier() {
199 static uint8_t multiplier = 100;
200 return multiplier;
201 }
202
203public:
209 DimmableLight(const __FlashStringHelper* name, uint8_t pin)
210 : SimpleLight(name, pin), _targetBrightness(0), _currentBrightness(0),
211 _lastBrightness(100), _lastUpdate(0), _lastMultiplier(100) {
212 const_cast<DeviceType&>(type) = DeviceType::LightDimmable;
213 }
214
222 virtual void setBrightness(uint8_t level) {
223 _targetBrightness = constrain(level, 0, 100);
224 if (_currentBrightness != _targetBrightness) {
225 _lastUpdate = millis();
226 }
227 if (_targetBrightness > 0) {
228 _lastBrightness = _targetBrightness;
229 }
230
231 _state = (_targetBrightness > 0);
232 EventSystem::instance().emit(EventType::DeviceValueChanged, this, _targetBrightness);
233 }
234
239 uint8_t getBrightness() const { return _targetBrightness; }
240
246 void update() override {
247 unsigned long now = millis();
248
249 // Check if multiplier changed (e.g., Night Mode toggled)
250 if (_lastMultiplier != brightness_multiplier()) {
251 _lastMultiplier = brightness_multiplier();
252 applyHardware(); // Immediately apply new multiplier
253 }
254
255 unsigned long diffTime = now - _lastUpdate;
256
257 if (diffTime >= MS_PER_STEP && _currentBrightness != _targetBrightness) {
258 uint8_t stepsToTake = diffTime / MS_PER_STEP;
259
260 if (_currentBrightness < _targetBrightness) {
261 uint8_t diff = _targetBrightness - _currentBrightness;
262 _currentBrightness += min(stepsToTake, diff);
263 } else {
264 uint8_t diff = _currentBrightness - _targetBrightness;
265 _currentBrightness -= min(stepsToTake, diff);
266 }
267
268 applyHardware();
269 _lastUpdate = now - (diffTime % MS_PER_STEP);
270 }
271 }
272
279 void toggle() override {
280 if (_state) {
281 _lastBrightness = _targetBrightness;
282 setBrightness(0);
283 } else {
284 setBrightness(_lastBrightness > 0 ? _lastBrightness : 100);
285 }
286 }
287
294 static void setBrightnessMultiplier(uint8_t multiplier) {
295 brightness_multiplier() = constrain(multiplier, 0, 100);
296 }
297
298protected:
302 void applyHardware() {
303 if (_currentBrightness == 0) {
304 analogWrite(_pin, PWM_MIN);
305 return;
306 }
307
308 // Apply brightness multiplier: (value * multiplier) / 100
309 uint16_t scaled = ((uint16_t)_currentBrightness * brightness_multiplier()) / 100;
310 uint8_t linear = map(scaled, 0, 100, PWM_MIN, PWM_MAX);
311
312 // Apply gamma correction
313 uint8_t gamma_corrected = pgm_read_byte(&GAMMA_LUT[linear]);
314 analogWrite(_pin, gamma_corrected);
315 }
