358 // FIX: First try to pass input to the current item

361 if (handled) {
362 _needs_redraw = true;

368 if (event == InputEvent::UP) {
369 if (_selected_index > 0) {
370 _selected_index--;
371
372 if (_selected_index < _scroll_offset) {

376 NavigationManager::instance().drawIncrementalCursor(oldIndex, _selected_index);
377 return true; // FIX: Handled by navigation

380 return true; // FIX: Handled (even if no movement)
381 } else if (event == InputEvent::DOWN) {

383 _selected_index++;

386 _scroll_offset = _selected_index - 2;

390 return true; // FIX: Handled by navigation

531

533 LCD_write_str(": ");
534
535 uint8_t val = (_device->*_getter)();
536 char buf[5];
537 itoa(val, buf, 10);
538 LCD_write_str(buf);
539
540 // Clear rest of first row
541 uint8_t cursor_pos = strlen_P((const char*)_label) + 2 + strlen(buf);
542 while (cursor_pos < 20) {
543 LCD_write_char(' ');
544 cursor_pos++;
545 }
546
547 // Row 1: Full-width progress bar (20 chars = 100 pixels)
548 LCD_set_cursor(0, row + 1);
549
550 // Map value to pixels (0-100 pixels across 20 chars)
551 uint16_t totalPixels = map(val, _min, _max, 0, 100);
552 uint8_t fullBlocks = totalPixels / 5;
553 uint8_t partialPixels = totalPixels % 5;
554
555 // Draw full blocks (0xFF = all 5 pixels filled)
556 for (uint8_t i = 0; i < fullBlocks && i < 20; i++) {
557 LCD_write_char(0xFF);
558 }
559
560 // Draw partial block
561 if (fullBlocks < 20 && partialPixels > 0) {
562 LCD_write_char(partialPixels - 1); // Custom char 0-4
563 fullBlocks++;
564 }
565
566 // Fill remaining with empty spaces
567 for (uint8_t i = fullBlocks; i < 20; i++) {
568 LCD_write_char(' ');
569 }
570 }

578 uint8_t current = (_device->*_getter)();
579
580 if (event == InputEvent::UP) {
581 // Safely increment with clamping
582 uint8_t newVal;
583 if (current > _max - _step) {
584 newVal = _max;
585 } else {
586 newVal = current + _step;
587 }

596 } else {

1408inline void NavigationManager::handleInput(InputEvent event) {
1409 MenuPage* current = getCurrentPage();
1410 if (!current) return;
1411
1412 if (event == InputEvent::BACK) {
1413 navigateBack();
1414 } else if (event == InputEvent::ENTER) {
1415 size_t idx = current->getSelectedIndex();
1416 if (idx < current->getItemsCount()) {
1417 MenuItem* item = current->getItem(idx);
1418 if (item->getType() == MenuItemType::SUBMENU) {
1419 SubMenuItem* sub = static_cast<SubMenuItem*>(item);
1420 MenuPage* newPage = sub->createPage();
1421 pushPage(newPage);
1422 } else {
1423 item->handleInput(event);
1424 }
1425 }
1426 } else {
1427 current->handleInput(event);
1428 }
1429
1430 if (getCurrentPage()) getCurrentPage()->forceRedraw();
1431}
