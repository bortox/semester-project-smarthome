275class ValueSliderItem : public MenuItem {
276private:
277 DeviceType* _device;
278 const __FlashStringHelper* _label;
279 uint8_t (DeviceType::*_getter)() const;
280 void (DeviceType::*_setter)(uint8_t);
281 uint8_t _min, _max, _step;
282
283 static bool _customCharsLoaded;
284
285 void loadCustomChars() {
286 if (!_customCharsLoaded) {
287 static const uint8_t customChars[] PROGMEM = {
288 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
289 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10, 0x10,
290 0x18, 0x18, 0x18, 0x18, 0x18, 0x18, 0x18, 0x18,
291 0x1C, 0x1C, 0x1C, 0x1C, 0x1C, 0x1C, 0x1C, 0x1C,
292 0x1E, 0x1E, 0x1E, 0x1E, 0x1E, 0x1E, 0x1E, 0x1E
293 };
294
295 uint8_t buffer[8];
296 for (uint8_t i = 0; i < 5; i++) {
297 memcpy_P(buffer, &customChars[i * 8], 8);
298 LCDcreateChar(i, buffer);
299 }
300 _customCharsLoaded = true;
301 }
302 }
303
304public:
315 ValueSliderItem(DeviceType* device,
316 const __FlashStringHelper* label,
317 uint8_t (DeviceType::*getter)() const,
318 void (DeviceType::*setter)(uint8_t),
319 uint8_t minVal, uint8_t maxVal, uint8_t step)
320 : _device(device), _label(label), _getter(getter), _setter(setter),
321 _min(minVal), _max(maxVal), _step(step) {
322 loadCustomChars();
323 }
324
325 bool relatesTo(IDevice* dev) override { return _device == dev; }
326
327 void draw(uint8_t row, bool selected) override {
328 LCD_set_cursor(0, row);
329
330 printLabel(_label);
331 LCD_write_str(": ");
332
333 uint8_t val = (_device->*_getter)();
334 char buf[5];
335 itoa(val, buf, 10);
336 LCD_write_str(buf);
337
338 uint8_t cursor_pos = strlen_P(reinterpret_cast<const char*>(_label)) + 2 + strlen(buf);
339 while (cursor_pos < 20) {
340 LCD_write_char(' ');
341 cursor_pos++;
342 }
343
344 LCD_set_cursor(0, row + 1);
345
346 uint16_t totalPixels = map(val, _min, _max, 0, 100);
347 uint8_t fullBlocks = totalPixels / 5;
348 uint8_t partialPixels = totalPixels % 5;
349
350 for (uint8_t i = 0; i < fullBlocks && i < 20; i++) {
351 LCD_write_char(0xFF);
352 }
353
354 if (fullBlocks < 20 && partialPixels > 0) {
355 LCD_write_char(partialPixels - 1);
356 fullBlocks++;
357 }
358
359 for (uint8_t i = fullBlocks; i < 20; i++) {
360 LCD_write_char(' ');
361 }
362 }
363
364 bool handleInput(InputEvent event) override {
365 uint8_t current = (_device->*_getter)();
366
367 if (event == InputEvent::UP) {
368 uint8_t newVal;
369 if (current > _max - _step) {
370 newVal = _max;
371 } else {
372 newVal = current + _step;
373 }
374 (_device->*_setter)(newVal);
375 return true;
376
377 } else if (event == InputEvent::DOWN) {
378 uint8_t newVal;
379 if (current < _min + _step) {
380 newVal = _min;
381 } else {
382 newVal = current - _step;
383 }
384 (_device->*_setter)(newVal);
385 return true;
386 }
387
388 return false;
389 }

429 void printValue(int16_t value) {
430 char buf[8];
431 if (_isTemperature) {
432 int16_t whole = value / 10;
433 int16_t decimal = abs(value % 10);
434
435 itoa(whole, buf, 10);
436 LCD_write_str(buf);
437 LCD_write_char('.');
438
439 buf[0] = '0' + decimal;
440 buf[1] = '\0';
441 LCD_write_str(buf);
442 LCD_write_char(0xDF);
443 } else {
444 itoa(value, buf, 10);
445 LCD_write_str(buf);
446 LCD_write_char(' ');
447 }
448 printLabel(_unit);
449 }
