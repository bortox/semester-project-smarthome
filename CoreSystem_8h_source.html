103 bool add(const T& item) {
104 if (_size >= _capacity) {
105 uint8_t newCapacity = _capacity + GROW_STEP;
106
107 if (newCapacity > MAX_CAPACITY) {
108 if (_capacity < MAX_CAPACITY) newCapacity = MAX_CAPACITY;
109 else return false;
110 }
111
112 T* newData = new T[newCapacity];
113 if (!newData) return false;
114
115 for (uint8_t i = 0; i < _size; i++) {
116 newData[i] = _data[i];
117 }
118
119 delete[] _data;
120 _data = newData;
121 _capacity = newCapacity;
122 }
123
124 _data[_size++] = item;
125 return true;
126 }
127
134 void remove(uint8_t index) {
135 if (index < _size) {
136 for (uint8_t i = index; i < _size - 1; i++) {
137 _data[i] = _data[i + 1];
138 }
139 _size--;
140
141 // NEW: Auto-shrink to save RAM when waste is significant
142 if (_capacity > SHRINK_THRESHOLD && (_capacity - _size) >= SHRINK_THRESHOLD) {
143 shrink();
144 }
145 }
146 }
147
148 // NEW: Manual memory reclamation
149 void shrink() {
150 if (_size == 0) {
151 delete[] _data;
152 _data = nullptr;
153 _capacity = 0;
154 return;
155 }
156
157 uint8_t newCapacity = ((_size / GROW_STEP) + 1) * GROW_STEP;
158 if (newCapacity >= _capacity) return;
159
160 T* newData = new T[newCapacity];
161 if (!newData) return;
162
163 for (uint8_t i = 0; i < _size; i++) {
164 newData[i] = _data[i];
165 }
166
167 delete[] _data;
168 _data = newData;
169 _capacity = newCapacity;
170 }
171
172 T& operator[](uint8_t index) { return _data[index]; }
173 const T& operator[](uint8_t index) const { return _data[index]; }
174 uint8_t size() const { return _size; }
175 void clear() { _size = 0; } // Nota: non libera memoria, resetta solo counter
