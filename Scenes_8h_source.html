151 void update() {
152 uint8_t count = _activeScenes.size();
153 if (count == 0) return;
154
155 // Bubble sort by priority (ascending)
156 for (uint8_t i = 0; i < count - 1; i++) {
157 for (uint8_t j = 0; j < count - i - 1; j++) {
158 if (_activeScenes[j]->getPriority() > _activeScenes[j + 1]->getPriority()) {
159 // Swap
160 IScene* temp = _activeScenes[j];
161 _activeScenes[j] = _activeScenes[j + 1];
162 _activeScenes[j + 1] = temp;
163 }
164 }
165 }
166
167 // Apply scenes in order (lowest priority first)
168 for (uint8_t i = 0; i < count; i++) {
169 _activeScenes[i]->update();
170 }
171 }

261 void update() override {
262 if (!_active) return;
263
264 unsigned long now = millis();
265 if (now - _lastChange >= CHANGE_INTERVAL_MS) {
266 _lastChange = now;

268 // Cycle color
269 _colorIndex = (_colorIndex + 1) % 3;
270
271 // Apply to all RGB lights
272 DeviceRegistry& registry = DeviceRegistry::instance();
273 DynamicArray<IDevice*>& devices = registry.getDevices();
274
275 for (uint8_t i = 0; i < devices.size(); i++) {
276 if (devices[i]->type == DeviceType::LightRGB) {
277 RGBLight* rgb = static_cast<RGBLight*>(devices[i]);
278
279 // Set color based on index
280 RGBColor color;
281 switch (_colorIndex) {
282 case 0: color = {255, 0, 0}; break; // Red
283 case 1: color = {0, 255, 0}; break; // Green
284 case 2: color = {0, 0, 255}; break; // Blue
285 }
286 rgb->setColor(color);
287
288 // Ensure brightness is on
289 if (rgb->getBrightness() == 0) {
290 rgb->setBrightness(100);
291 }
