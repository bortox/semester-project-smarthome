145class VccSensor : public Sensor<int16_t> {
146public:
147 VccSensor(const __FlashStringHelper* name) : Sensor<int16_t>(name) {}
148
153 int16_t getValue() const override {
154 // Save current ADMUX state
155 uint8_t savedADMUX = ADMUX;
156
157 // Set reference to AVcc and measure internal 1.1V bandgap
158 #if defined(__AVR_ATmega32U4__)
159 // ATmega32U4: MUX[4:0] = 11110 for bandgap
160 ADMUX = _BV(REFS0) | _BV(MUX4) | _BV(MUX3) | _BV(MUX2) | _BV(MUX1);
161 #else
162 // ATmega328P: MUX[3:0] = 1110 for bandgap
163 ADMUX = _BV(REFS0) | _BV(MUX3) | _BV(MUX2) | _BV(MUX1);
164 #endif
165
166 // Wait for voltage to stabilize
167 delayMicroseconds(250);
168
169 // Start conversion
170 ADCSRA |= _BV(ADSC);
171
172 // Wait for conversion to complete
173 while (bit_is_set(ADCSRA, ADSC));
174
175 // Read ADC value (low byte first for atomic read)
176 uint8_t low = ADCL;
177 uint8_t high = ADCH;
178 uint16_t adcValue = (high << 8) | low;
179
180 // Restore ADMUX
181 ADMUX = savedADMUX;
182
183 // Calculate VCC: VCC = 1.1V * 1024 / ADC
184 // Using 1125300L = 1.1 * 1023 * 1000 for millivolt result
185 if (adcValue == 0) return 0;
186 return (int16_t)(1125300L / adcValue);
187 }
188};
